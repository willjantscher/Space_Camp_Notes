using System;
using System.Collections.Generic;
using System.Linq;
using UI.CommonControls.Grid.Adapters;
using UI.Filter;
using UnityEngine;
using UnityEngine.UI;

namespace UI.CommonControls.Grid.Custom
{
    public class SatelliteFilterHeaderRow : HeaderRow
    {
        [SerializeField] private DataColumnHeader _satIdHeaderCell = default;
        [SerializeField] private DataColumnHeader _satNameHeaderCell = default;
        [SerializeField] private DataColumnHeader _satNicknameHeaderCell = default;
        [SerializeField] private DataColumnHeader _satRegimeHeaderCell = default;
        [SerializeField] private DataColumnHeader _selectedHeaderCell = default;
        [SerializeField] private Toggle _selectedToggle = default;

        private SatFilterUI _satFilter;
        private AssetListEditor.AssetListEditor _assetListEditor;

        public override void Initialize(DataGrid parentGrid)
        {
            base.Initialize(parentGrid);

            _selectedHeaderCell.OnHeaderClicked += SortByColumn;
            _selectedHeaderCell.GetValueFunc = GetSelectedHeaderCell;
            _selectedHeaderCell.Initialize();
            _satIdHeaderCell.OnHeaderClicked += SortByColumn;
            _satIdHeaderCell.GetValueFunc = GetSatIdHeaderCell;
            _satIdHeaderCell.Initialize();
            _satNameHeaderCell.OnHeaderClicked += SortByColumn;
            _satNameHeaderCell.GetValueFunc = GetSatNameHeaderCell;
            _satNameHeaderCell.Initialize();
            _satNicknameHeaderCell.OnHeaderClicked += SortByColumn;
            _satNicknameHeaderCell.GetValueFunc = GetSatNicknameHeaderCell;
            _satNicknameHeaderCell.Initialize();
            _satRegimeHeaderCell.OnHeaderClicked += SortByColumn;
            _satRegimeHeaderCell.GetValueFunc = GetSatRegimeHeaderCell;
            _satRegimeHeaderCell.Initialize();

            _selectedToggle.onValueChanged.AddListener(HandleSelectAllValueChanged);

            ParentGrid.OnDataSourceChanged += HandleGridDataSourceChanged;

            _satFilter = transform.root.gameObject.GetComponentInChildren<SatFilterUI>();
            _assetListEditor = transform.root.gameObject.GetComponentInChildren<AssetListEditor.AssetListEditor>();
        }

        private IComparable GetSatRegimeHeaderCell(IDataRowViewModel arg)
        {
            return (arg as SatelliteFilterDataRowViewModel)?.Regime ?? string.Empty;
        }

        private IComparable GetSatNicknameHeaderCell(IDataRowViewModel arg)
        {
            return (arg as SatelliteFilterDataRowViewModel)?.SatNickname ?? string.Empty;
        }

        private IComparable GetSatNameHeaderCell(IDataRowViewModel arg)
        {
            return (arg as SatelliteFilterDataRowViewModel)?.SatName ?? string.Empty;
        }

        private IComparable GetSatIdHeaderCell(IDataRowViewModel arg)
        {
            string value = (arg as SatelliteFilterDataRowViewModel)?.SatNo ?? string.Empty;
            return value.PadLeft(5, '0');
        }

        private IComparable GetSelectedHeaderCell(IDataRowViewModel arg)
        {
            return arg is SatelliteFilterDataRowViewModel model && model.IsSelected;
        }

        public void HandleHeaderRowSelectAll(bool isSelected)
        {
            if (isSelected) HandleSelectAll();
            else HandleDeselectAll();
        }

        private void HandleSelectAll()
        {
            int currentSelected = _satFilter.SelectedAssets.Count;
            foreach (IDataRowViewModel dataRow in ParentGrid.DataSource)
            {
                if (currentSelected >= _assetListEditor.MaxSatCount) break;
                if (!(dataRow is SatelliteFilterDataRowViewModel dataModel) || dataModel.IsSelected)
                {
                    continue;
                }
                // ReSharper disable once DelegateSubtraction
                dataModel.OnPropertyChanged -= HandleViewModelChanged;
                dataModel.IsSelected = true;
                dataModel.OnPropertyChanged += HandleViewModelChanged;
                currentSelected += 1;
            }

            _selectedToggle.onValueChanged.RemoveListener(HandleSelectAllValueChanged);
            _selectedToggle.isOn = true;
            _selectedToggle.onValueChanged.AddListener(HandleSelectAllValueChanged);
        }

        private void HandleDeselectAll()
        {
            List<SatelliteFilterDataRowViewModel> selectedDataRows = _satFilter.GetFilteredViewModels().OfType<SatelliteFilterDataRowViewModel>().ToList().Where(r => r.IsSelected == true).ToList();

            foreach (IDataRowViewModel dataRow in selectedDataRows)
            {
                if (!(dataRow is SatelliteFilterDataRowViewModel dataModel))
                {
                    continue;
                }
                // ReSharper disable once DelegateSubtraction
                dataModel.OnPropertyChanged -= HandleViewModelChanged;
                dataModel.IsSelected = false;
                dataModel.OnPropertyChanged += HandleViewModelChanged;
            }
        }

        private void HandleGridDataSourceChanged()
        {
            foreach (SatelliteFilterDataRowViewModel viewModel in ParentGrid.DataSource.Select(da => da as SatelliteFilterDataRowViewModel))
            {
                if (viewModel != null)
                {
                    viewModel.OnPropertyChanged += HandleViewModelChanged;
                }
            }

            _selectedToggle.onValueChanged.RemoveListener(HandleSelectAllValueChanged);
            HandleHeaderSelectedToggle();
            _selectedToggle.onValueChanged.AddListener(HandleSelectAllValueChanged);
        }

        private void HandleViewModelChanged(SatelliteFilterDataRowViewModel obj)
        {
            _selectedToggle.onValueChanged.RemoveListener(HandleSelectAllValueChanged);
            HandleHeaderSelectedToggle();
            _selectedToggle.onValueChanged.AddListener(HandleSelectAllValueChanged);
        }

        private void HandleHeaderSelectedToggle()
        {
            List<SatelliteFilterDataRowViewModel> filteredRows = _satFilter.GetFilteredViewModels().OfType<SatelliteFilterDataRowViewModel>().ToList();

            bool allAssetsSelectedUnderMax = filteredRows.Count < _assetListEditor.MaxSatCount &&
                filteredRows.Where(r => r.IsSelected == true).ToList().Count == filteredRows.Count;
            bool maxSelected = filteredRows.Where(r => r.IsSelected == true).ToList().Count == _assetListEditor.MaxSatCount;

            _selectedToggle.isOn = allAssetsSelectedUnderMax || maxSelected;
        }

        private void HandleSelectAllValueChanged(bool isSelected)
        {
            if (ParentGrid.DataSource == null)
            {
                return;
            }

            HandleHeaderRowSelectAll(isSelected);
        }
    }
}
